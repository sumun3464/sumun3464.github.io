def combination(n,m):
    if n==m or m==0:
        return 1
    return combination(n-1, m-1)+combination(n-1,m)

n,m = map(int, input().split())
num = combination(n,m)
count = 0

while num%10==0:
    count += 1
    num = num//10

print(count)
